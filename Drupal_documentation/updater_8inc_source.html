00308   public function prepareInstallDirectory(&$filetransfer, $directory) {
00309     // Make the parent dir writable if need be and create the dir.
00310     if (!is_dir($directory)) {
00311       $parent_dir = dirname($directory);
00312       if (!is_writable($parent_dir)) {
00313         @chmod($parent_dir, 0755);
00314         // It is expected that this will fail if the directory is owned by the
00315         // FTP user. If the FTP user == web server, it will succeed.
00316         try {
00317           $filetransfer->createDirectory($directory);
00318           $this->makeWorldReadable($filetransfer, $directory);
00319         }
00320         catch (FileTransferException $e) {
00321           // Probably still not writable. Try to chmod and do it again.
00322           // @todo: Make a new exception class so we can catch it differently.
00323           try {
00324             $old_perms = substr(sprintf('%o', fileperms($parent_dir)), -4);
00325             $filetransfer->chmod($parent_dir, 0755);
