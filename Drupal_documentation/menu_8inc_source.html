01210 function menu_tree_page_data($menu_name, $max_depth = NULL, $only_active_trail = FALSE) {
01211   $tree = &drupal_static(__FUNCTION__, array());
01212 
01213   // Check if the active trail has been overridden for this menu tree.
01214   $active_path = menu_tree_get_path($menu_name);
01215   // Load the menu item corresponding to the current page.
01216   if ($item = menu_get_item($active_path)) {
01217     if (isset($max_depth)) {
01218       $max_depth = min($max_depth, MENU_MAX_DEPTH);
01219     }
01220     // Generate a cache ID (cid) specific for this page.
01221     $cid = 'links:' . $menu_name . ':page:' . $item['href'] . ':' . $GLOBALS['language']->language . ':' . (int) $item['access'] . ':' . (int) $max_depth;
01222     // If we are asked for the active trail only, and $menu_name has not been
01223     // built and cached for this page yet, then this likely means that it
01224     // won't be built anymore, as this function is invoked from
01225     // template_process_page(). So in order to not build a giant menu tree
01226     // that needs to be checked for access on all levels, we simply check
01227     // whether we have the menu already in cache, or otherwise, build a minimum
01228     // tree containing the breadcrumb/active trail only.
01229     // @see menu_set_active_trail()
01230     if (!isset($tree[$cid]) && $only_active_trail) {
01231       $cid .= ':trail';
01232     }
01233 
01234     if (!isset($tree[$cid])) {
01235       // If the static variable doesn't have the data, check {cache_menu}.
01236       $cache = cache_get($cid, 'cache_menu');
01237       if ($cache && isset($cache->data)) {
01238         // If the cache entry exists, it contains the parameters for
01239         // menu_build_tree().
01240         $tree_parameters = $cache->data;
01241       }
01242       // If the tree data was not in the cache, build $tree_parameters.
01243       if (!isset($tree_parameters)) {
01244         $tree_parameters = array(
01245           'min_depth' => 1,
01246           'max_depth' => $max_depth,
01247         );
01248         // Parent mlids; used both as key and value to ensure uniqueness.
01249         // We always want all the top-level links with plid == 0.
01250         $active_trail = array(0 => 0);
01251 
01252         // If the item for the current page is accessible, build the tree
01253         // parameters accordingly.
01254         if ($item['access']) {
01255           // Find a menu link corresponding to the current path. If $active_path
01256           // is NULL, let menu_link_get_preferred() determine the path.

01259             // active trail, if it resides in the requested menu. Otherwise,
01260             // we'd needlessly re-run _menu_build_tree() queries for every menu
01261             // on every page.
01262             if ($active_link['menu_name'] == $menu_name) {
01263               // Use all the coordinates, except the last one because there
01264               // can be no child beyond the last column.
01265               for ($i = 1; $i < MENU_MAX_DEPTH; $i++) {
01266                 if ($active_link['p' . $i]) {
01267                   $active_trail[$active_link['p' . $i]] = $active_link['p' . $i];
01268                 }
01269               }
01270               // If we are asked to build links for the active trail only, skip
01271               // the entire 'expanded' handling.
01272               if ($only_active_trail) {
01273                 $tree_parameters['only_active_trail'] = TRUE;
01274               }
01275             }
01276           }
01277           $parents = $active_trail;
01278 
01279           $expanded = variable_get('menu_expanded', array());
01280           // Check whether the current menu has any links set to be expanded.
01281           if (!$only_active_trail && in_array($menu_name, $expanded)) {
01282             // Collect all the links set to be expanded, and then add all of
01283             // their children to the list as well.
01284             do {
01285               $result = db_select('menu_links', NULL, array('fetch' => PDO::FETCH_ASSOC))
01286                 ->fields('menu_links', array('mlid'))
01287                 ->condition('menu_name', $menu_name)
01288                 ->condition('expanded', 1)
01289                 ->condition('has_children', 1)
01290                 ->condition('plid', $parents, 'IN')
01291                 ->condition('mlid', $parents, 'NOT IN')
01292                 ->execute();
01293               $num_rows = FALSE;
01294               foreach ($result as $item) {
01295                 $parents[$item['mlid']] = $item['mlid'];
01296                 $num_rows = TRUE;
01297               }
01298             } while ($num_rows);
01299           }
01300           $tree_parameters['expanded'] = $parents;
01301           $tree_parameters['active_trail'] = $active_trail;
01302         }
01303         // If access is denied, we only show top-level links in menus.
01304         else {
01305           $tree_parameters['expanded'] = $active_trail;
01306           $tree_parameters['active_trail'] = $active_trail;
01307         }
01308         // Cache the tree building parameters using the page-specific cid.
01309         cache_set($cid, $tree_parameters, 'cache_menu');
01310       }
01311 
01312       // Build the tree using the parameters; the resulting tree will be cached
01313       // by _menu_build_tree().
01314       $tree[$cid] = menu_build_tree($menu_name, $tree_parameters);
01315     }
01316     return $tree[$cid];
01317   }

01364 function _menu_build_tree($menu_name, array $parameters = array()) {
01365   // Static cache of already built menu trees.
01366   $trees = &drupal_static(__FUNCTION__, array());
01367 
01368   // Build the cache id; sort parents to prevent duplicate storage and remove
01369   // default parameter values.
01370   if (isset($parameters['expanded'])) {
01371     sort($parameters['expanded']);
01372   }
01373   $tree_cid = 'links:' . $menu_name . ':tree-data:' . $GLOBALS['language']->language . ':' . hash('sha256', serialize($parameters));
01374 
01375   // If we do not have this tree in the static cache, check {cache_menu}.
01376   if (!isset($trees[$tree_cid])) {
01377     $cache = cache_get($tree_cid, 'cache_menu');
01378     if ($cache && isset($cache->data)) {
01379       $trees[$tree_cid] = $cache->data;
01380     }
01381   }
01382 
01383   if (!isset($trees[$tree_cid])) {
01384     // Select the links from the table, and recursively build the tree. We
01385     // LEFT JOIN since there is no match in {menu_router} for an external
01386     // link.
01387     $query = db_select('menu_links', 'ml', array('fetch' => PDO::FETCH_ASSOC));
01388     $query->addTag('translatable');
01389     $query->leftJoin('menu_router', 'm', 'm.path = ml.router_path');
01390     $query->fields('ml');
01391     $query->fields('m', array(
01392       'load_functions',
01393       'to_arg_functions',
01394       'access_callback',
01395       'access_arguments',
01396       'page_callback',
01397       'page_arguments',
01398       'delivery_callback',
01399       'tab_parent',
01400       'tab_root',
01401       'title',
01402       'title_callback',
01403       'title_arguments',
01404       'theme_callback',
01405       'theme_arguments',
01406       'type',
01407       'description',
01408     ));
01409     for ($i = 1; $i <= MENU_MAX_DEPTH; $i++) {
01410       $query->orderBy('p' . $i, 'ASC');
01411     }
01412     $query->condition('ml.menu_name', $menu_name);
01413     if (!empty($parameters['expanded'])) {
01414       $query->condition('ml.plid', $parameters['expanded'], 'IN');
01415     }
01416     elseif (!empty($parameters['only_active_trail'])) {
01417       $query->condition('ml.mlid', $parameters['active_trail'], 'IN');
01418     }
01419     $min_depth = (isset($parameters['min_depth']) ? $parameters['min_depth'] : 1);
01420     if ($min_depth != 1) {
01421       $query->condition('ml.depth', $min_depth, '>=');
01422     }
01423     if (isset($parameters['max_depth'])) {
01424       $query->condition('ml.depth', $parameters['max_depth'], '<=');
01425     }
01426     // Add custom query conditions, if any were passed.
01427     if (isset($parameters['conditions'])) {
01428       foreach ($parameters['conditions'] as $column => $value) {
01429         $query->condition($column, $value);
01430       }
01431     }
01432 
01433     // Build an ordered array of links using the query result object.
01434     $links = array();
01435     foreach ($query->execute() as $item) {
01436       $links[] = $item;
01437     }
01438     $active_trail = (isset($parameters['active_trail']) ? $parameters['active_trail'] : array());
01439     $data['tree'] = menu_tree_data($links, $active_trail, $min_depth);
01440     $data['node_links'] = array();
01441     menu_tree_collect_node_links($data['tree'], $data['node_links']);
01442 
01443     // Cache the data, if it is not already in the cache.
01444     cache_set($tree_cid, $data, 'cache_menu');
01445     $trees[$tree_cid] = $data;
01446   }
01447 
01448   return $trees[$tree_cid];
01449 }

01562 function _menu_tree_data(&$links, $parents, $depth) {
01563   $tree = array();
01564   while ($item = array_pop($links)) {
01565     // We need to determine if we're on the path to root so we can later build
01566     // the correct active trail and breadcrumb.
01567     $item['in_active_trail'] = in_array($item['mlid'], $parents);
01568     // Add the current link to the tree.
01569     $tree[$item['mlid']] = array(
01570       'link' => $item,
01571       'below' => array(),
01572     );
01573     // Look ahead to the next link, but leave it on the array so it's available
01574     // to other recursive function calls if we return or build a sub-tree.
01575     $next = end($links);
01576     // Check whether the next link is the first in a new sub-tree.
01577     if ($next && $next['depth'] > $depth) {
01578       // Recursively call _menu_tree_data to build the sub-tree.
01579       $tree[$item['mlid']]['below'] = _menu_tree_data($links, $parents, $next['depth']);
01580       // Fetch next link after filling the sub-tree.
01581       $next = end($links);
01582     }
01583     // Determine if we should exit the loop and return.
01584     if (!$next || $next['depth'] < $depth) {
01585       break;
01586     }
01587   }
01588   return $tree;
01589 }

01830 function menu_navigation_links($menu_name, $level = 0) {
01831   // Don't even bother querying the menu table if no menu is specified.
01832   if (empty($menu_name)) {
01833     return array();
01834   }
01835 
01836   // Get the menu hierarchy for the current page.
01837   $tree = menu_tree_page_data($menu_name, $level + 1);
01838 
01839   // Go down the active trail until the right level is reached.
01840   while ($level-- > 0 && $tree) {
01841     // Loop through the current level's items until we find one that is in trail.
01842     while ($item = array_shift($tree)) {
01843       if ($item['link']['in_active_trail']) {
01844         // If the item is in the active trail, we continue in the subtree.
01845         $tree = empty($item['below']) ? array() : $item['below'];
01846         break;
01847       }
01848     }
01849   }
01850 
01851   // Create a single level of links.
01852   $router_item = menu_get_item();
01853   $links = array();
01854   foreach ($tree as $item) {
01855     if (!$item['link']['hidden']) {
01856       $class = '';
01857       $l = $item['link']['localized_options'];
01858       $l['href'] = $item['link']['href'];
01859       $l['title'] = $item['link']['title'];
01860       if ($item['link']['in_active_trail']) {
01861         $class = ' active-trail';
01862         $l['attributes']['class'][] = 'active-trail';
01863       }
01864       // Normally, l() compares the href of every link with $_GET['q'] and sets
01865       // the active class accordingly. But local tasks do not appear in menu

01895 function menu_local_tasks($level = 0) {
01896   $data = &drupal_static(__FUNCTION__);
01897   $root_path = &drupal_static(__FUNCTION__ . ':root_path', '');
01898   $empty = array(
01899     'tabs' => array('count' => 0, 'output' => array()),
01900     'actions' => array('count' => 0, 'output' => array()),
01901     'root_path' => &$root_path,
01902   );
01903 
01904   if (!isset($data)) {
01905     $data = array();
01906     // Set defaults in case there are no actions or tabs.
01907     $actions = $empty['actions'];
01908     $tabs = array();
01909 
01910     $router_item = menu_get_item();
01911 
01912     // If this router item points to its parent, start from the parents to
01913     // compute tabs and actions.
01914     if ($router_item && ($router_item['type'] & MENU_LINKS_TO_PARENT)) {
01915       $router_item = menu_get_item($router_item['tab_parent_href']);
01916     }
01917 
01918     // If we failed to fetch a router item or the current user doesn't have
01919     // access to it, don't bother computing the tabs.
01920     if (!$router_item || !$router_item['access']) {
01921       return $empty;
01922     }
01923 
01924     // Get all tabs (also known as local tasks) and the root page.
01925     $result = db_select('menu_router', NULL, array('fetch' => PDO::FETCH_ASSOC))
01926       ->fields('menu_router')
01927       ->condition('tab_root', $router_item['tab_root'])
01928       ->condition('context', MENU_CONTEXT_INLINE, '<>')
01929       ->orderBy('weight')
01930       ->orderBy('title')
01931       ->execute();
01932     $map = $router_item['original_map'];
01933     $children = array();
01934     $tasks = array();
01935     $root_path = $router_item['path'];
01936 
01937     foreach ($result as $item) {
01938       _menu_translate($item, $map, TRUE);
01939       if ($item['tab_parent']) {
01940         // All tabs, but not the root page.
01941         $children[$item['tab_parent']][$item['path']] = $item;
01942       }
01943       // Store the translated item for later use.
01944       $tasks[$item['path']] = $item;
01945     }
01946 
01947     // Find all tabs below the current path.
01948     $path = $router_item['path'];
01949     // Tab parenting may skip levels, so the number of parts in the path may not
01950     // equal the depth. Thus we use the $depth counter (offset by 1000 for ksort).
01951     $depth = 1001;
01952     $actions['count'] = 0;
01953     $actions['output'] = array();
01954     while (isset($children[$path])) {
01955       $tabs_current = array();
01956       $actions_current = array();
01957       $next_path = '';
01958       $tab_count = 0;
01959       $action_count = 0;
01960       foreach ($children[$path] as $item) {
01961         // Local tasks can be normal items too, so bitmask with
01962         // MENU_IS_LOCAL_TASK before checking.
01963         if (!($item['type'] & MENU_IS_LOCAL_TASK)) {
01964           // This item is not a tab, skip it.
01965           continue;
01966         }
01967         if ($item['access']) {
01968           $link = $item;
01969           // The default task is always active. As tabs can be normal items
01970           // too, so bitmask with MENU_LINKS_TO_PARENT before checking.
01971           if (($item['type'] & MENU_LINKS_TO_PARENT) == MENU_LINKS_TO_PARENT) {
01972             // Find the first parent which is not a default local task or action.
01973             for ($p = $item['tab_parent']; ($tasks[$p]['type'] & MENU_LINKS_TO_PARENT) == MENU_LINKS_TO_PARENT; $p = $tasks[$p]['tab_parent']);
01974             // Use the path of the parent instead.
01975             $link['href'] = $tasks[$p]['href'];
01976             // Mark the link as active, if the current path happens to be the
01977             // path of the default local task itself (i.e., instead of its
01978             // tab_parent_href or tab_root_href). Normally, links for default
01979             // local tasks link to their parent, but the path of default local
01980             // tasks can still be accessed directly, in which case this link
01981             // would not be marked as active, since l() only compares the href
01982             // with $_GET['q'].
01983             if ($link['href'] != $_GET['q']) {
01984               $link['localized_options']['attributes']['class'][] = 'active';
01985             }
01986             $tabs_current[] = array(
01987               '#theme' => 'menu_local_task',
01988               '#link' => $link,
01989               '#active' => TRUE,
01990             );
01991             $next_path = $item['path'];
01992             $tab_count++;
01993           }
01994           else {
01995             // Actions can be normal items too, so bitmask with
01996             // MENU_IS_LOCAL_ACTION before checking.
01997             if (($item['type'] & MENU_IS_LOCAL_ACTION) == MENU_IS_LOCAL_ACTION) {
01998               // The item is an action, display it as such.
01999               $actions_current[] = array(
02000                 '#theme' => 'menu_local_action',
02001                 '#link' => $link,
02002               );
02003               $action_count++;
02004             }
02005             else {
02006               // Otherwise, it's a normal tab.
02007               $tabs_current[] = array(
02008                 '#theme' => 'menu_local_task',
02009                 '#link' => $link,
02010               );
02011               $tab_count++;
02012             }
02013           }
02014         }
02015       }
02016       $path = $next_path;
02017       $tabs[$depth]['count'] = $tab_count;
02018       $tabs[$depth]['output'] = $tabs_current;
02019       $actions['count'] += $action_count;
02020       $actions['output'] = array_merge($actions['output'], $actions_current);
02021       $depth++;
02022     }
02023     $data['actions'] = $actions;
02024     // Find all tabs at the same level or above the current one.
02025     $parent = $router_item['tab_parent'];
02026     $path = $router_item['path'];
02027     $current = $router_item;
02028     $depth = 1000;
02029     while (isset($children[$parent])) {
02030       $tabs_current = array();
02031       $next_path = '';
02032       $next_parent = '';
02033       $count = 0;
02034       foreach ($children[$parent] as $item) {
02035         // Skip local actions.
02036         if ($item['type'] & MENU_IS_LOCAL_ACTION) {
02037           continue;
02038         }
02039         if ($item['access']) {
02040           $count++;
02041           $link = $item;
02042           // Local tasks can be normal items too, so bitmask with
02043           // MENU_LINKS_TO_PARENT before checking.
02044           if (($item['type'] & MENU_LINKS_TO_PARENT) == MENU_LINKS_TO_PARENT) {
02045             // Find the first parent which is not a default local task.
02046             for ($p = $item['tab_parent']; ($tasks[$p]['type'] & MENU_LINKS_TO_PARENT) == MENU_LINKS_TO_PARENT; $p = $tasks[$p]['tab_parent']);
02047             // Use the path of the parent instead.
02048             $link['href'] = $tasks[$p]['href'];
02049             if ($item['path'] == $router_item['path']) {
02050               $root_path = $tasks[$p]['path'];
02051             }
02052           }
02053           // We check for the active tab.
02054           if ($item['path'] == $path) {
02055             // Mark the link as active, if the current path is a (second-level)
02056             // local task of a default local task. Since this default local task
02057             // links to its parent, l() will not mark it as active, as it only
02058             // compares the link's href to $_GET['q'].
02059             if ($link['href'] != $_GET['q']) {
02060               $link['localized_options']['attributes']['class'][] = 'active';
02061             }
02062             $tabs_current[] = array(
02063               '#theme' => 'menu_local_task',
02064               '#link' => $link,
02065               '#active' => TRUE,
02066             );
02067             $next_path = $item['tab_parent'];
02068             if (isset($tasks[$next_path])) {
02069               $next_parent = $tasks[$next_path]['tab_parent'];
02070             }
02071           }
02072           else {
02073             $tabs_current[] = array(
02074               '#theme' => 'menu_local_task',
02075               '#link' => $link,
02076             );
02077           }
02078         }
02079       }
02080       $path = $next_path;
02081       $parent = $next_parent;
02082       $tabs[$depth]['count'] = $count;
02083       $tabs[$depth]['output'] = $tabs_current;
02084       $depth--;
02085     }
02086     // Sort by depth.
02087     ksort($tabs);
02088     // Remove the depth, we are interested only in their relative placement.
02089     $tabs = array_values($tabs);
02090     $data['tabs'] = $tabs;
02091 
02092     // Allow modules to alter local tasks or dynamically append further tasks.
02093     drupal_alter('menu_local_tasks', $data, $router_item, $root_path);
02094   }
02095 
02096   if (isset($data['tabs'][$level])) {
02097     return array(
02098       'tabs' => $data['tabs'][$level],
02099       'actions' => $data['actions'],
02100       'root_path' => $root_path,
02101     );
02102   }
02103   // @todo If there are no tabs, then there still can be actions; for example,
02104   //   when added via hook_menu_local_tasks_alter().
02105   elseif (!empty($data['actions']['output'])) {
02106     return array('actions' => $data['actions']) + $empty;
02107   }
02108   return $empty;
02109 }

02148 function menu_contextual_links($module, $parent_path, $args) {
02149   static $path_empty = array();
02150 
02151   $links = array();
02152   // Performance: In case a previous invocation for the same parent path did not
02153   // return any links, we immediately return here.
02154   if (isset($path_empty[$parent_path]) && strpos($parent_path, '%') !== FALSE) {
02155     return $links;
02156   }
02157   // Construct the item-specific parent path.
02158   $path = $parent_path . '/' . implode('/', $args);
02159 
02160   // Get the router item for the given parent link path.
02161   $router_item = menu_get_item($path);
02162   if (!$router_item || !$router_item['access']) {
02163     $path_empty[$parent_path] = TRUE;
02164     return $links;
02165   }
02166   $data = &drupal_static(__FUNCTION__, array());
02167   $root_path = $router_item['path'];
02168 
02169   // Performance: For a single, normalized path (such as 'node/%') we only query
02170   // available tasks once per request.
02171   if (!isset($data[$root_path])) {
02172     // Get all contextual links that are direct children of the router item and
02173     // not of the tab type 'view'.
02174     $data[$root_path] = db_select('menu_router', 'm')
02175       ->fields('m')
02176       ->condition('tab_parent', $router_item['tab_root'])
02177       ->condition('context', MENU_CONTEXT_NONE, '<>')
02178       ->condition('context', MENU_CONTEXT_PAGE, '<>')
02179       ->orderBy('weight')
02180       ->orderBy('title')
02181       ->execute()
02182       ->fetchAllAssoc('path', PDO::FETCH_ASSOC);
02183   }
02184   $parent_length = drupal_strlen($root_path) + 1;
02185   $map = $router_item['original_map'];
02186   foreach ($data[$root_path] as $item) {
02187     // Extract the actual "task" string from the path argument.
02188     $key = drupal_substr($item['path'], $parent_length);
02189 
02190     // Denormalize and translate the contextual link.
02191     _menu_translate($item, $map, TRUE);
02192     if (!$item['access']) {
02193       continue;
02194     }
02195     // All contextual links are keyed by the actual "task" path argument,
02196     // prefixed with the name of the implementing module.
02197     $links[$module . '-' . $key] = $item;
02198   }
02199 
02200   // Allow modules to alter contextual links.
02201   drupal_alter('menu_contextual_links', $links, $router_item, $root_path);
02202 
02203   // Performance: If the current user does not have access to any links for this
02204   // router path and no other module added further links, we assign FALSE here
02205   // to skip the entire process the next time the same router path is requested.
02206   if (empty($links)) {
02207     $path_empty[$parent_path] = TRUE;
02208   }
02209 
02210   return $links;
02211 }

02348 function menu_set_active_trail($new_trail = NULL) {
02349   $trail = &drupal_static(__FUNCTION__);
02350 
02351   if (isset($new_trail)) {
02352     $trail = $new_trail;
02353   }
02354   elseif (!isset($trail)) {
02355     $trail = array();
02356     $trail[] = array(
02357       'title' => t('Home'),
02358       'href' => '<front>',
02359       'link_path' => '',
02360       'localized_options' => array(),
02361       'type' => 0,
02362     );
02363 
02364     // Try to retrieve a menu link corresponding to the current path. If more
02365     // than one exists, the link from the most preferred menu is returned.
02366     $preferred_link = menu_link_get_preferred();
02367     $current_item = menu_get_item();
02368 
02369     // There is a link for the current path.
02370     if ($preferred_link) {

02373       // the given menu has not been built in this request yet.

02376     }
02377     // There is no link for the current path.
02378     else {
02379       $preferred_link = $current_item;
02380       $curr = FALSE;
02381     }
02382

02384       $link = $curr['link'];
02385       if ($link['in_active_trail']) {

02389           // that have not been translated yet, since they contain dynamic
02390           // argument placeholders (%). Such links are not contained in regular
02391           // menu trees, and have only been loaded for the additional
02392           // translation that happens here, so as to be able to display them in
02393           // the breadcumb for the current page.
02394           // @see _menu_tree_check_access()
02395           // @see _menu_link_translate()
02396           if (strpos($link['href'], '%') !== FALSE) {
02397             _menu_link_translate($link, TRUE);
02398           }
02399           if ($link['access']) {
02400             $trail[] = $link;
02401           }
02402         }

02404       }

02406     }
02407     // Make sure the current page is in the trail to build the page title, by
02408     // appending either the preferred link or the menu router item for the

02411     if ($preferred_link && $last['href'] != $preferred_link['href'] && !drupal_is_front_page()) {

02413     }
02414   }

02435 function menu_link_get_preferred($path = NULL, $selected_menu = NULL) {
02436   $preferred_links = &drupal_static(__FUNCTION__);
02437 
02438   if (!isset($path)) {
02439     $path = $_GET['q'];
02440   }
02441 
02442   if (empty($selected_menu)) {
02443     // Use an illegal menu name as the key for the preferred menu link.
02444     $selected_menu = MENU_PREFERRED_LINK;
02445   }
02446 
02447   if (!isset($preferred_links[$path])) {
02448     // Look for the correct menu link by building a list of candidate paths,
02449     // which are ordered by priority (translated hrefs are preferred over
02450     // untranslated paths). Afterwards, the most relevant path is picked from
02451     // the menus, ordered by menu preference.
02452     $item = menu_get_item($path);
02453     $path_candidates = array();
02454     // 1. The current item href.
02455     $path_candidates[$item['href']] = $item['href'];
02456     // 2. The tab root href of the current item (if any).
02457     if ($item['tab_parent'] && ($tab_root = menu_get_item($item['tab_root_href']))) {
02458       $path_candidates[$tab_root['href']] = $tab_root['href'];
02459     }
02460     // 3. The current item path (with wildcards).
02461     $path_candidates[$item['path']] = $item['path'];
02462     // 4. The tab root path of the current item (if any).
02463     if (!empty($tab_root)) {
02464       $path_candidates[$tab_root['path']] = $tab_root['path'];
02465     }
02466 
02467     // Retrieve a list of menu names, ordered by preference.
02468     $menu_names = menu_get_active_menu_names();
02469     // Put the selected menu at the front of the list.
02470     array_unshift($menu_names, $selected_menu);
02471 
02472     $query = db_select('menu_links', 'ml', array('fetch' => PDO::FETCH_ASSOC));
02473     $query->leftJoin('menu_router', 'm', 'm.path = ml.router_path');
02474     $query->fields('ml');
02475     // Weight must be taken from {menu_links}, not {menu_router}.
02476     $query->addField('ml', 'weight', 'link_weight');
02477     $query->fields('m');
02478     $query->condition('ml.link_path', $path_candidates, 'IN');
02479 
02480     // Sort candidates by link path and menu name.
02481     $candidates = array();
02482     foreach ($query->execute() as $candidate) {
02483       $candidate['weight'] = $candidate['link_weight'];
02484       $candidates[$candidate['link_path']][$candidate['menu_name']] = $candidate;
02485       // Add any menus not already in the menu name search list.
02486       if (!in_array($candidate['menu_name'], $menu_names)) {
02487         $menu_names[] = $candidate['menu_name'];
02488       }
02489     }
02490 
02491     // Store the most specific link for each menu. Also save the most specific
02492     // link of the most preferred menu in $preferred_link.
02493     foreach ($path_candidates as $link_path) {
02494       if (isset($candidates[$link_path])) {
02495         foreach ($menu_names as $menu_name) {
02496           if (empty($preferred_links[$path][$menu_name]) && isset($candidates[$link_path][$menu_name])) {
02497             $candidate_item = $candidates[$link_path][$menu_name];
02498             $map = explode('/', $path);
02499             _menu_translate($candidate_item, $map);
02500             if ($candidate_item['access']) {
02501               $preferred_links[$path][$menu_name] = $candidate_item;
02502               if (empty($preferred_links[$path][MENU_PREFERRED_LINK])) {
02503                 // Store the most specific link.
02504                 $preferred_links[$path][MENU_PREFERRED_LINK] = $candidate_item;
02505               }
02506             }
02507           }
02508         }
02509       }
02510     }
02511   }
02512 
02513   return isset($preferred_links[$path][$selected_menu]) ? $preferred_links[$path][$selected_menu] : FALSE;
02514 }

02543 function menu_get_active_breadcrumb() {
02544   $breadcrumb = array();
02545 
02546   // No breadcrumb for the front page.
02547   if (drupal_is_front_page()) {
02548     return $breadcrumb;
02549   }
02550 
02551   $item = menu_get_item();
02552   if (!empty($item['access'])) {
02553     $active_trail = menu_get_active_trail();
02554 
02555     // Allow modules to alter the breadcrumb, if possible, as that is much
02556     // faster than rebuilding an entirely new active trail.
02557     drupal_alter('menu_breadcrumb', $active_trail, $item);
02558 
02559     // Don't show a link to the current page in the breadcrumb trail.
02560     $end = end($active_trail);
02561     if ($item['href'] == $end['href']) {
02562       array_pop($active_trail);
02563     }
02564 
02565     // Remove the tab root (parent) if the current path links to its parent.
02566     // Normally, the tab root link is included in the breadcrumb, as soon as we
02567     // are on a local task or any other child link. However, if we are on a
02568     // default local task (e.g., node/%/view), then we do not want the tab root
02569     // link (e.g., node/%) to appear, as it would be identical to the current
02570     // page. Since this behavior also needs to work recursively (i.e., on
02571     // default local tasks of default local tasks), and since the last non-task
02572     // link in the trail is used as page title (see menu_get_active_title()),
02573     // this condition cannot be cleanly integrated into menu_get_active_trail().
02574     // menu_get_active_trail() already skips all links that link to their parent
02575     // (commonly MENU_DEFAULT_LOCAL_TASK). In order to also hide the parent link
02576     // itself, we always remove the last link in the trail, if the current
02577     // router item links to its parent.
02578     if (($item['type'] & MENU_LINKS_TO_PARENT) == MENU_LINKS_TO_PARENT) {
02579       array_pop($active_trail);
02580     }
02581 
02582     foreach ($active_trail as $parent) {
02583       $breadcrumb[] = l($parent['title'], $parent['href'], $parent['localized_options']);
02584     }
02585   }
02586   return $breadcrumb;

02799 function _menu_navigation_links_rebuild($menu) {
02800   // Add normal and suggested items as links.
02801   $menu_links = array();
02802   foreach ($menu as $path => $item) {
02803     if ($item['_visible']) {
02804       $menu_links[$path] = $item;
02805       $sort[$path] = $item['_number_parts'];
02806     }
02807   }
02808   if ($menu_links) {
02809     // Keep an array of processed menu links, to allow menu_link_save() to
02810     // check this for parents instead of querying the database.
02811     $parent_candidates = array();
02812     // Make sure no child comes before its parent.
02813     array_multisort($sort, SORT_NUMERIC, $menu_links);
02814 
02815     foreach ($menu_links as $key => $item) {
02816       $existing_item = db_select('menu_links')
02817         ->fields('menu_links')
02818         ->condition('link_path', $item['path'])
02819         ->condition('module', 'system')
02820         ->execute()->fetchAssoc();
02821       if ($existing_item) {
02822         $item['mlid'] = $existing_item['mlid'];
02823         // A change in hook_menu may move the link to a different menu
02824         if (empty($item['menu_name']) || ($item['menu_name'] == $existing_item['menu_name'])) {
02825           $item['menu_name'] = $existing_item['menu_name'];
02826           $item['plid'] = $existing_item['plid'];
02827         }
02828         else {
02829           // It moved to a new menu. Let menu_link_save() try to find a new
02830           // parent based on the path.
02831           unset($item['plid']);
02832         }
02833         $item['has_children'] = $existing_item['has_children'];
02834         $item['updated'] = $existing_item['updated'];
02835       }
02836       if ($existing_item && $existing_item['customized']) {
02837         $parent_candidates[$existing_item['mlid']] = $existing_item;
02838       }
02839       else {
02840         $item = _menu_link_build($item);
02841         menu_link_save($item, $existing_item, $parent_candidates);
02842         $parent_candidates[$item['mlid']] = $item;
02843         unset($menu_links[$key]);
02844       }
02845     }
02846   }
02847   $paths = array_keys($menu);
02848   // Updated and customized items whose router paths are gone need new ones.
02849   $result = db_select('menu_links', NULL, array('fetch' => PDO::FETCH_ASSOC))
02850     ->fields('menu_links', array(
02851       'link_path',
02852       'mlid',
02853       'router_path',
02854       'updated',
02855     ))
02856     ->condition(db_or()
02857       ->condition('updated', 1)
02858       ->condition(db_and()
02859         ->condition('router_path', $paths, 'NOT IN')
02860         ->condition('external', 0)
02861         ->condition('customized', 1)
02862       )
02863     )
02864     ->execute();
02865   foreach ($result as $item) {
02866     $router_path = _menu_find_router_path($item['link_path']);
02867     if (!empty($router_path) && ($router_path != $item['router_path'] || $item['updated'])) {
02868       // If the router path and the link path matches, it's surely a working
02869       // item, so we clear the updated flag.
02870       $updated = $item['updated'] && $router_path != $item['link_path'];
02871       db_update('menu_links')
02872         ->fields(array(
02873           'router_path' => $router_path,
02874           'updated' => (int) $updated,
02875         ))
02876         ->condition('mlid', $item['mlid'])
02877         ->execute();
02878     }
02879   }
02880   // Find any item whose router path does not exist any more.
02881   $result = db_select('menu_links')
02882     ->fields('menu_links')
02883     ->condition('router_path', $paths, 'NOT IN')
02884     ->condition('external', 0)
02885     ->condition('updated', 0)
02886     ->condition('customized', 0)
02887     ->orderBy('depth', 'DESC')
02888     ->execute();
02889   // Remove all such items. Starting from those with the greatest depth will
02890   // minimize the amount of re-parenting done by menu_link_delete().
02891   foreach ($result as $item) {
02892     _menu_delete_item($item, TRUE);
02893   }
02894 }

03051 function menu_link_save(&$item, $existing_item = array(), $parent_candidates = array()) {
03052   drupal_alter('menu_link', $item);
03053 
03054   // This is the easiest way to handle the unique internal path '<front>',
03055   // since a path marked as external does not need to match a router path.
03056   $item['external'] = (url_is_external($item['link_path'])  || $item['link_path'] == '<front>') ? 1 : 0;
03057   // Load defaults.
03058   $item += array(
03059     'menu_name' => 'navigation',
03060     'weight' => 0,
03061     'link_title' => '',
03062     'hidden' => 0,
03063     'has_children' => 0,
03064     'expanded' => 0,
03065     'options' => array(),
03066     'module' => 'menu',
03067     'customized' => 0,
03068     'updated' => 0,
03069   );
03070   if (isset($item['mlid'])) {
03071     if (!$existing_item) {
03072       $existing_item = db_query('SELECT * FROM {menu_links} WHERE mlid = :mlid', array('mlid' => $item['mlid']))->fetchAssoc();
03073     }
03074     if ($existing_item) {
03075       $existing_item['options'] = unserialize($existing_item['options']);
03076     }
03077   }
03078   else {
03079     $existing_item = FALSE;
03080   }
03081 
03082   // Try to find a parent link. If found, assign it and derive its menu.
03083   $parent = _menu_link_find_parent($item, $parent_candidates);
03084   if (!empty($parent['mlid'])) {
03085     $item['plid'] = $parent['mlid'];
03086     $item['menu_name'] = $parent['menu_name'];
03087   }
03088   // If no corresponding parent link was found, move the link to the top-level.
03089   else {
03090     $item['plid'] = 0;
03091   }
03092   $menu_name = $item['menu_name'];
03093 
03094   if (!$existing_item) {
03095     $item['mlid'] = db_insert('menu_links')
03096       ->fields(array(
03097         'menu_name' => $item['menu_name'],
03098         'plid' => $item['plid'],
03099         'link_path' => $item['link_path'],
03100         'hidden' => $item['hidden'],
03101         'external' => $item['external'],
03102         'has_children' => $item['has_children'],
03103         'expanded' => $item['expanded'],
03104         'weight' => $item['weight'],
03105         'module' => $item['module'],
03106         'link_title' => $item['link_title'],
03107         'options' => serialize($item['options']),
03108         'customized' => $item['customized'],
03109         'updated' => $item['updated'],
03110       ))
03111       ->execute();
03112   }
03113 
03114   // Directly fill parents for top-level links.
03115   if ($item['plid'] == 0) {
03116     $item['p1'] = $item['mlid'];
03117     for ($i = 2; $i <= MENU_MAX_DEPTH; $i++) {
03118       $item["p$i"] = 0;
03119     }
03120     $item['depth'] = 1;
03121   }
03122   // Otherwise, ensure that this link's depth is not beyond the maximum depth
03123   // and fill parents based on the parent link.
03124   else {
03125     if ($item['has_children'] && $existing_item) {
03126       $limit = MENU_MAX_DEPTH - menu_link_children_relative_depth($existing_item) - 1;
03127     }
03128     else {
03129       $limit = MENU_MAX_DEPTH - 1;
03130     }
03131     if ($parent['depth'] > $limit) {
03132       return FALSE;
03133     }
03134     $item['depth'] = $parent['depth'] + 1;
03135     _menu_link_parents_set($item, $parent);
03136   }
03137   // Need to check both plid and menu_name, since plid can be 0 in any menu.
03138   if ($existing_item && ($item['plid'] != $existing_item['plid'] || $menu_name != $existing_item['menu_name'])) {
03139     _menu_link_move_children($item, $existing_item);
03140   }
03141   // Find the router_path.
03142   if (empty($item['router_path'])  || !$existing_item || ($existing_item['link_path'] != $item['link_path'])) {
03143     if ($item['external']) {

03145     }
03146     else {
03147       // Find the router path which will serve this path.
03148       $item['parts'] = explode('/', $item['link_path'], MENU_MAX_PARTS);
03149       $item['router_path'] = _menu_find_router_path($item['link_path']);
03150     }
03151   }

03155   // $item may have additional keys left over from building a router entry.
03156   // The intersect removes the extra keys, allowing a meaningful comparison.
03157   if (!$existing_item || (array_intersect_key($item, $existing_item) != $existing_item)) {
03158     db_update('menu_links')
03159       ->fields(array(
03160         'menu_name' => $item['menu_name'],
03161         'plid' => $item['plid'],
03162         'link_path' => $item['link_path'],
03163         'router_path' => $item['router_path'],
03164         'hidden' => $item['hidden'],
03165         'external' => $item['external'],
03166         'has_children' => $item['has_children'],
03167         'expanded' => $item['expanded'],
03168         'weight' => $item['weight'],
03169         'depth' => $item['depth'],
03170         'p1' => $item['p1'],
03171         'p2' => $item['p2'],
03172         'p3' => $item['p3'],
03173         'p4' => $item['p4'],
03174         'p5' => $item['p5'],
03175         'p6' => $item['p6'],
03176         'p7' => $item['p7'],
03177         'p8' => $item['p8'],
03178         'p9' => $item['p9'],
03179         'module' => $item['module'],
03180         'link_title' => $item['link_title'],
03181         'options' => serialize($item['options']),
03182         'customized' => $item['customized'],
03183       ))
03184       ->condition('mlid', $item['mlid'])
03185       ->execute();

03187     _menu_update_parental_status($item);
03188     menu_cache_clear($menu_name);
03189     if ($existing_item && $menu_name != $existing_item['menu_name']) {
03190       menu_cache_clear($existing_item['menu_name']);
03191     }
03192     // Notify modules we have acted on a menu item.
03193     $hook = 'menu_link_insert';
03194     if ($existing_item) {

03223 function _menu_link_find_parent($menu_link, $parent_candidates = array()) {
03224   $parent = FALSE;
03225 
03226   // This item is explicitely top-level, skip the rest of the parenting.
03227   if (isset($menu_link['plid']) && empty($menu_link['plid'])) {
03228     return $parent;
03229   }
03230 
03231   // If we have a parent link ID, try to use that.
03232   $candidates = array();
03233   if (isset($menu_link['plid'])) {
03234     $candidates[] = $menu_link['plid'];
03235   }
03236 
03237   // Else, if we have a link hierarchy try to find a valid parent in there.
03238   if (!empty($menu_link['depth']) && $menu_link['depth'] > 1) {
03239     for ($depth = $menu_link['depth'] - 1; $depth >= 1; $depth--) {
03240       $candidates[] = $menu_link['p' . $depth];
03241     }
03242   }
03243 
03244   foreach ($candidates as $mlid) {
03245     if (isset($parent_candidates[$mlid])) {
03246       $parent = $parent_candidates[$mlid];
03247     }
03248     else {
03249       $parent = db_query("SELECT * FROM {menu_links} WHERE mlid = :mlid", array(':mlid' => $mlid))->fetchAssoc();
03250     }
03251     if ($parent) {
03252       return $parent;
03253     }
03254   }
03255 
03256   // If everything else failed, try to derive the parent from the path
03257   // hierarchy. This only makes sense for links derived from menu router
03258   // items (ie. from hook_menu()).
03259   if ($menu_link['module'] == 'system') {
03260     $query = db_select('menu_links');
03261     $query->condition('module', 'system');
03262     // We always respect the link's 'menu_name'; inheritance for router items is
03263     // ensured in _menu_router_build().
03264     $query->condition('menu_name', $menu_link['menu_name']);
03265 
03266     // Find the parent - it must be unique.
03267     $parent_path = $menu_link['link_path'];
03268     do {
03269       $parent = FALSE;
03270       $parent_path = substr($parent_path, 0, strrpos($parent_path, '/'));
03271       $new_query = clone $query;
03272       $new_query->condition('link_path', $parent_path);
03273       // Only valid if we get a unique result.
03274       if ($new_query->countQuery()->execute()->fetchField() == 1) {
03275         $parent = $new_query->fields('menu_links')->execute()->fetchAssoc();
03276       }
03277     } while ($parent === FALSE && $parent_path);
03278   }
03279 
03280   return $parent;
03281 }

03323 function _menu_find_router_path($link_path) {
03324   // $menu will only have data during a menu rebuild.
03325   $menu = _menu_router_cache();
03326 
03327   $router_path = $link_path;
03328   $parts = explode('/', $link_path, MENU_MAX_PARTS);
03329   $ancestors = menu_get_ancestors($parts);
03330 
03331   if (empty($menu)) {
03332     // Not during a menu rebuild, so look up in the database.
03333     $router_path = (string) db_select('menu_router')
03334       ->fields('menu_router', array('path'))
03335       ->condition('path', $ancestors, 'IN')
03336       ->orderBy('fit', 'DESC')
03337       ->range(0, 1)
03338       ->execute()->fetchField();
03339   }

03521 function _menu_router_build($callbacks) {
03522   // First pass: separate callbacks from paths, making paths ready for
03523   // matching. Calculate fitness, and fill some default values.
03524   $menu = array();
03525   $masks = array();
03526   foreach ($callbacks as $path => $item) {
03527     $load_functions = array();
03528     $to_arg_functions = array();
03529     $fit = 0;
03530     $move = FALSE;
03531 
03532     $parts = explode('/', $path, MENU_MAX_PARTS);
03533     $number_parts = count($parts);
03534     // We store the highest index of parts here to save some work in the fit
03535     // calculation loop.
03536     $slashes = $number_parts - 1;
03537     // Extract load and to_arg functions.
03538     foreach ($parts as $k => $part) {
03539       $match = FALSE;
03540       // Look for wildcards in the form allowed to be used in PHP functions,
03541       // because we are using these to construct the load function names.
03542       if (preg_match('/^%(|' . DRUPAL_PHP_FUNCTION_PATTERN . ')$/', $part, $matches)) {
03543         if (empty($matches[1])) {
03544           $match = TRUE;
03545           $load_functions[$k] = NULL;
03546         }
03547         else {
03548           if (function_exists($matches[1] . '_to_arg')) {
03549             $to_arg_functions[$k] = $matches[1] . '_to_arg';
03550             $load_functions[$k] = NULL;
03551             $match = TRUE;
03552           }
03553           if (function_exists($matches[1] . '_load')) {
03554             $function = $matches[1] . '_load';
03555             // Create an array of arguments that will be passed to the _load
03556             // function when this menu path is checked, if 'load arguments'
03557             // exists.
03558             $load_functions[$k] = isset($item['load arguments']) ? array($function => $item['load arguments']) : $function;
03559             $match = TRUE;
03560           }
03561         }
03562       }
03563       if ($match) {
03564         $parts[$k] = '%';
03565       }
03566       else {
03567         $fit |=  1 << ($slashes - $k);
03568       }
03569     }
03570     if ($fit) {
03571       $move = TRUE;
03572     }
03573     else {
03574       // If there is no %, it fits maximally.
03575       $fit = (1 << $number_parts) - 1;
03576     }
03577     $masks[$fit] = 1;
03578     $item['_load_functions'] = $load_functions;
03579     $item['to_arg_functions'] = empty($to_arg_functions) ? '' : serialize($to_arg_functions);
03580     $item += array(
03581       'title' => '',
03582       'weight' => 0,
03583       'type' => MENU_NORMAL_ITEM,
03584       'module' => '',
03585       '_number_parts' => $number_parts,
03586       '_parts' => $parts,
03587       '_fit' => $fit,
03588     );
03589     $item += array(
03590       '_visible' => (bool) ($item['type'] & MENU_VISIBLE_IN_BREADCRUMB),
03591       '_tab' => (bool) ($item['type'] & MENU_IS_LOCAL_TASK),
03592     );
03593     if ($move) {
03594       $new_path = implode('/', $item['_parts']);
03595       $menu[$new_path] = $item;
03596       $sort[$new_path] = $number_parts;
03597     }
03598     else {
03599       $menu[$path] = $item;
03600       $sort[$path] = $number_parts;
03601     }
03602   }
03603   array_multisort($sort, SORT_NUMERIC, $menu);
03604   // Apply inheritance rules.
03605   foreach ($menu as $path => $v) {
03606     $item = &$menu[$path];
03607     if (!$item['_tab']) {
03608       // Non-tab items.
03609       $item['tab_parent'] = '';
03610       $item['tab_root'] = $path;
03611     }
03612     // If not specified, assign the default tab type for local tasks.
03613     elseif (!isset($item['context'])) {
03614       $item['context'] = MENU_CONTEXT_PAGE;
03615     }
03616     for ($i = $item['_number_parts'] - 1; $i; $i--) {
03617       $parent_path = implode('/', array_slice($item['_parts'], 0, $i));
03618       if (isset($menu[$parent_path])) {
03619 
03620         $parent = &$menu[$parent_path];
03621 
03622         // If we have no menu name, try to inherit it from parent items.
03623         if (!isset($item['menu_name'])) {
03624           // If the parent item of this item does not define a menu name (and no
03625           // previous iteration assigned one already), try to find the menu name
03626           // of the parent item in the currently stored menu links.
03627           if (!isset($parent['menu_name'])) {
03628             $menu_name = db_query("SELECT menu_name FROM {menu_links} WHERE router_path = :router_path AND module = 'system'", array(':router_path' => $parent_path))->fetchField();
03629             if ($menu_name) {
03630               $parent['menu_name'] = $menu_name;
03631             }
03632           }
03633           // If the parent item defines a menu name, inherit it.
03634           if (!empty($parent['menu_name'])) {
03635             $item['menu_name'] = $parent['menu_name'];
03636           }
03637         }
03638         if (!isset($item['tab_parent'])) {
03639           // Parent stores the parent of the path.
03640           $item['tab_parent'] = $parent_path;
03641         }
03642         if (!isset($item['tab_root']) && !$parent['_tab']) {
03643           $item['tab_root'] = $parent_path;
03644         }
03645         // If an access callback is not found for a default local task we use
03646         // the callback from the parent, since we expect them to be identical.
03647         // In all other cases, the access parameters must be specified.
03648         if (($item['type'] == MENU_DEFAULT_LOCAL_TASK) && !isset($item['access callback']) && isset($parent['access callback'])) {
03649           $item['access callback'] = $parent['access callback'];
03650           if (!isset($item['access arguments']) && isset($parent['access arguments'])) {
03651             $item['access arguments'] = $parent['access arguments'];
03652           }
03653         }
03654         // Same for page callbacks.
03655         if (!isset($item['page callback']) && isset($parent['page callback'])) {

03661             $item['file path'] = $parent['file path'];

03667             }
03668           }
03669         }
03670         // Same for delivery callbacks.
03671         if (!isset($item['delivery callback']) && isset($parent['delivery callback'])) {
03672           $item['delivery callback'] = $parent['delivery callback'];
03673         }
03674         // Same for theme callbacks.
03675         if (!isset($item['theme callback']) && isset($parent['theme callback'])) {
03676           $item['theme callback'] = $parent['theme callback'];
03677           if (!isset($item['theme arguments']) && isset($parent['theme arguments'])) {

03679           }
03680         }
03681         // Same for load arguments: if a loader doesn't have any explict
03682         // arguments, try to find arguments in the parent.
03683         if (!isset($item['load arguments'])) {
03684           foreach ($item['_load_functions'] as $k => $function) {

03686             if (!is_array($function)) {

03688               // using the same function name and defining arguments...

03692               }
03693             }
03694           }
03695         }
03696       }
03697     }
03698     if (!isset($item['access callback']) && isset($item['access arguments'])) {
03699       // Default callback.
03700       $item['access callback'] = 'user_access';
03701     }
03702     if (!isset($item['access callback']) || empty($item['page callback'])) {

03704     }
03705     if (is_bool($item['access callback'])) {
03706       $item['access callback'] = intval($item['access callback']);
03707     }
03708 
03709     $item['load_functions'] = empty($item['_load_functions']) ? '' : serialize($item['_load_functions']);
03710     $item += array(
03711       'access arguments' => array(),
03712       'access callback' => '',
03713       'page arguments' => array(),
03714       'page callback' => '',
03715       'delivery callback' => '',
03716       'title arguments' => array(),
03717       'title callback' => 't',
