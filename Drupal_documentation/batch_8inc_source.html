00391 function _batch_api_percentage($total, $current) {
00392   if (!$total || $total == $current) {
00393     // If $total doesn't evaluate as true or is equal to the current set, then
00394     // we're finished, and we can return "100".
00395     $percentage = "100";
00396   }
00397   else {
00398     // We add a new digit at 200, 2000, etc. (since, for example, 199/200
00399     // would round up to 100% if we didn't).
00400     $decimal_places = max(0, floor(log10($total / 2.0)) - 1);
00401     do {
00402       // Calculate the percentage to the specified number of decimal places.
00403       $percentage = sprintf('%01.' . $decimal_places . 'f', round($current / $total * 100, $decimal_places));
00404       // When $current is an integer, the above calculation will always be
00405       // correct. However, if $current is a floating point number (in the case
00406       // of a multi-step batch operation that is not yet complete), $percentage
00407       // may be erroneously rounded up to 100%. To prevent that, we add one
00408       // more decimal place and try again.
00409       $decimal_places++;
00410     } while ($percentage == '100');
00411   }
00412   return $percentage;
