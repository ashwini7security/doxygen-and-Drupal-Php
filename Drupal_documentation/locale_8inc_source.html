01247 function _locale_import_parse_arithmetic($string) {
01248   // Operator precedence table
01249   $precedence = array("(" => -1, ")" => -1, "?" => 1, ":" => 1, "||" => 3, "&&" => 4, "==" => 5, "!=" => 5, "<" => 6, ">" => 6, "<=" => 6, ">=" => 6, "+" => 7, "-" => 7, "*" => 8, "/" => 8, "%" => 8);
01250   // Right associativity
01251   $right_associativity = array("?" => 1, ":" => 1);
01252 
01253   $tokens = _locale_import_tokenize_formula($string);
01254 
01255   // Parse by converting into infix notation then back into postfix
01256   // Operator stack - holds math operators and symbols
01257   $operator_stack = array();
01258   // Element Stack - holds data to be operated on
01259   $element_stack = array();
01260 
01261   foreach ($tokens as $token) {
01262     $current_token = $token;
01263 
01264     // Numbers and the $n variable are simply pushed into $element_stack
01265     if (is_numeric($token)) {
01266       $element_stack[] = $current_token;
01267     }
01268     elseif ($current_token == "n") {
01269       $element_stack[] = '$n';
01270     }
01271     elseif ($current_token == "(") {
01272       $operator_stack[] = $current_token;
01273     }
01274     elseif ($current_token == ")") {
01275       $topop = array_pop($operator_stack);
01276       while (isset($topop) && ($topop != "(")) {
01277         $element_stack[] = $topop;
01278         $topop = array_pop($operator_stack);
01279       }
01280     }
01281     elseif (!empty($precedence[$current_token])) {
01282       // If it's an operator, then pop from $operator_stack into $element_stack until the
01283       // precedence in $operator_stack is less than current, then push into $operator_stack
01284       $topop = array_pop($operator_stack);
01285       while (isset($topop) && ($precedence[$topop] >= $precedence[$current_token]) && !(($precedence[$topop] == $precedence[$current_token]) && !empty($right_associativity[$topop]) && !empty($right_associativity[$current_token]))) {
01286         $element_stack[] = $topop;
01287         $topop = array_pop($operator_stack);
01288       }
01289       if ($topop) {
01290         $operator_stack[] = $topop;   // Return element to top
01291       }
01292       $operator_stack[] = $current_token;      // Parentheses are not needed
01293     }
01294     else {
01295       return FALSE;
01296     }
01297   }
01298 
01299   // Flush operator stack
01300   $topop = array_pop($operator_stack);
01301   while ($topop != NULL) {
01302     $element_stack[] = $topop;
01303     $topop = array_pop($operator_stack);
01304   }
01305 
01306   // Now extract formula from stack
01307   $previous_size = count($element_stack) + 1;
01308   while (count($element_stack) < $previous_size) {
01309     $previous_size = count($element_stack);
01310     for ($i = 2; $i < count($element_stack); $i++) {
01311       $op = $element_stack[$i];
01312       if (!empty($precedence[$op])) {
01313         $f = "";
01314         if ($op == ":") {
01315           $f = $element_stack[$i - 2] . "):" . $element_stack[$i - 1] . ")";
01316         }
01317         elseif ($op == "?") {
01318           $f = "(" . $element_stack[$i - 2] . "?(" . $element_stack[$i - 1];
01319         }
01320         else {
01321           $f = "(" . $element_stack[$i - 2] . $op . $element_stack[$i - 1] . ")";
01322         }
01323         array_splice($element_stack, $i - 2, 3, $f);
01324         break;
01325       }
01326     }
01327   }
01328 
01329   // If only one element is left, the number of operators is appropriate
01330   if (count($element_stack) == 1) {
01331     return $element_stack[0];
01332   }
01333   else {
01334     return FALSE;
01335   }
01336 }

01347 function _locale_import_tokenize_formula($formula) {
01348   $formula = str_replace(" ", "", $formula);
01349   $tokens = array();
01350   for ($i = 0; $i < strlen($formula); $i++) {
01351     if (is_numeric($formula[$i])) {
01352       $num = $formula[$i];
01353       $j = $i + 1;
01354       while ($j < strlen($formula) && is_numeric($formula[$j])) {
01355         $num .= $formula[$j];
01356         $j++;
01357       }
01358       $i = $j - 1;
01359       $tokens[] = $num;
01360     }
01361     elseif ($pos = strpos(" =<>!&|", $formula[$i])) { // We won't have a space
01362       $next = $formula[$i + 1];
01363       switch ($pos) {
01364         case 1:
01365         case 2:
01366         case 3:
01367         case 4:
01368           if ($next == '=') {
01369             $tokens[] = $formula[$i] . '=';
01370             $i++;
01371           }
01372           else {
01373             $tokens[] = $formula[$i];
01374           }
01375           break;
01376         case 5:
01377           if ($next == '&') {
01378             $tokens[] = '&&';
01379             $i++;
01380           }
01381           else {
01382             $tokens[] = $formula[$i];
01383           }
01384           break;
01385         case 6:
01386           if ($next == '|') {
01387             $tokens[] = '||';
01388             $i++;
01389           }
01390           else {
01391             $tokens[] = $formula[$i];
01392           }
01393           break;
01394       }
01395     }
01396     else {
01397       $tokens[] = $formula[$i];
01398     }
01399   }
01400   return $tokens;
01401 }
