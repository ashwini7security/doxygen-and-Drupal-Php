01563 function form_set_error($name = NULL, $message = '', $limit_validation_errors = NULL) {
01564   $form = &drupal_static(__FUNCTION__, array());
01565   $sections = &drupal_static(__FUNCTION__ . ':limit_validation_errors');
01566   if (isset($limit_validation_errors)) {
01567     $sections = $limit_validation_errors;
01568   }

01570   if (isset($name) && !isset($form[$name])) {
01571     $record = TRUE;
01572     if (isset($sections)) {
01573       // #limit_validation_errors is an array of "sections" within which user
01574       // input must be valid. If the element is within one of these sections,
01575       // the error must be recorded. Otherwise, it can be suppressed.
01576       // #limit_validation_errors can be an empty array, in which case all
01577       // errors are suppressed. For example, a "Previous" button might want its
01578       // submit action to be triggered even if none of the submitted values are
01579       // valid.

01582         // Exploding by '][' reconstructs the element's #parents. If the
01583         // reconstructed #parents begin with the same keys as the specified
01584         // section, then the element's values are within the part of
01585         // $form_state['values'] that the clicked button requires to be valid,
01586         // so errors for this element must be recorded. As the exploded array
01587         // will all be strings, we need to cast every value of the section
01588         // array to string.
01589         if (array_slice(explode('][', $name), 0, count($section)) === array_map('strval', $section)) {
01590           $record = TRUE;
01591           break;
01592         }
01593       }
01594     }
01595     if ($record) {
01596       $form[$name] = $message;
01597       if ($message) {
01598         drupal_set_message($message, 'error');
01599       }
01600     }
01601   }
01602 
01603   return $form;
01604 }

01896       }
01897     }

01901     if (!empty($form_state['triggering_element']['#executes_submit_callback'])) {

01903     }

01908       // have been determined (including from input variables set by JavaScript

01915       $form_state['values'][$form_state['triggering_element']['#name']] = $form_state['triggering_element']['#value'];
01916 
01917       // @todo Legacy support. Remove in Drupal 8.

01919     }
01920   }

01927 function _form_builder_handle_input_element($form_id, &$element, &$form_state) {
01928   if (!isset($element['#name'])) {
01929     $name = array_shift($element['#parents']);
01930     $element['#name'] = $name;
01931     if ($element['#type'] == 'file') {
01932       // To make it easier to handle $_FILES in file.inc, we place all
01933       // file fields in the 'files' array. Also, we do not support
01934       // nested file names.
01935       $element['#name'] = 'files[' . $element['#name'] . ']';
01936     }
01937     elseif (count($element['#parents'])) {
01938       $element['#name'] .= '[' . implode('][', $element['#parents']) . ']';
01939     }
01940     array_unshift($element['#parents'], $name);
01941   }
01942 
01943   // Setting #disabled to TRUE results in user input being ignored, regardless
01944   // of how the element is themed or whether JavaScript is used to change the
01945   // control's attributes. However, it's good UI to let the user know that input
01946   // is not wanted for the control. HTML supports two attributes for this:
01947   // http://www.w3.org/TR/html401/interact/forms.html#h-17.12. If a form wants
01948   // to start a control off with one of these attributes for UI purposes only,
01949   // but still allow input to be processed if it's sumitted, it can set the
01950   // desired attribute in #attributes directly rather than using #disabled.
01951   // However, developers should think carefully about the accessibility
01952   // implications of doing so: if the form expects input to be enterable under
01953   // some condition triggered by JavaScript, how would someone who has
01954   // JavaScript disabled trigger that condition? Instead, developers should
01955   // consider whether a multi-step form would be more appropriate (#disabled can
01956   // be changed from step to step). If one still decides to use JavaScript to
01957   // affect when a control is enabled, then it is best for accessibility for the
01958   // control to be enabled in the HTML, and disabled by JavaScript on document
01959   // ready.
01960   if (!empty($element['#disabled'])) {
01961     if (!empty($element['#allow_focus'])) {
01962       $element['#attributes']['readonly'] = 'readonly';
01963     }
01964     else {
01965       $element['#attributes']['disabled'] = 'disabled';
01966     }
01967   }
01968 
01969   // With JavaScript or other easy hacking, input can be submitted even for
01970   // elements with #access=FALSE or #disabled=TRUE. For security, these must
01971   // not be processed. Forms that set #disabled=TRUE on an element do not
01972   // expect input for the element, and even forms submitted with
01973   // drupal_form_submit() must not be able to get around this. Forms that set
01974   // #access=FALSE on an element usually allow access for some users, so forms
01975   // submitted with drupal_form_submit() may bypass access restriction and be
01976   // treated as high-privilege users instead.
01977   $process_input = empty($element['#disabled']) && ($form_state['programmed'] || ($form_state['process_input'] && (!isset($element['#access']) || $element['#access'])));
01978 
01979   // Set the element's #value property.
01980   if (!isset($element['#value']) && !array_key_exists('#value', $element)) {
01981     $value_callback = !empty($element['#value_callback']) ? $element['#value_callback'] : 'form_type_' . $element['#type'] . '_value';
01982     if ($process_input) {
01983       // Get the input for the current element. NULL values in the input need to
01984       // be explicitly distinguished from missing input. (see below)
01985       $input_exists = NULL;
01986       $input = drupal_array_get_nested_value($form_state['input'], $element['#parents'], $input_exists);
01987       // For browser-submitted forms, the submitted values do not contain values
01988       // for certain elements (empty multiple select, unchecked checkbox).
01989       // During initial form processing, we add explicit NULL values for such
01990       // elements in $form_state['input']. When rebuilding the form, we can
01991       // distinguish elements having NULL input from elements that were not part
01992       // of the initially submitted form and can therefore use default values
01993       // for the latter, if required. Programmatically submitted forms can
01994       // submit explicit NULL values when calling drupal_form_submit(), so we do
01995       // not modify $form_state['input'] for them.
01996       if (!$input_exists && !$form_state['rebuild'] && !$form_state['programmed']) {
01997         // Add the necessary parent keys to $form_state['input'] and sets the
01998         // element's input value to NULL.
01999         drupal_array_set_nested_value($form_state['input'], $element['#parents'], NULL);
02000         $input_exists = TRUE;
02001       }
02002       // If we have input for the current element, assign it to the #value
02003       // property, optionally filtered through $value_callback.
02004       if ($input_exists) {
02005         if (function_exists($value_callback)) {
02006           $element['#value'] = $value_callback($element, $input, $form_state);
02007         }
02008         if (!isset($element['#value']) && isset($input)) {
02009           $element['#value'] = $input;
02010         }
02011       }
02012       // Mark all posted values for validation.
02013       if (isset($element['#value']) || (!empty($element['#required']))) {
02014         $element['#needs_validation'] = TRUE;
02015       }
02016     }
02017     // Load defaults.
02018     if (!isset($element['#value'])) {
02019       // Call #type_value without a second argument to request default_value handling.
02020       if (function_exists($value_callback)) {
02021         $element['#value'] = $value_callback($element, FALSE, $form_state);
02022       }
02023       // Final catch. If we haven't set a value yet, use the explicit default value.
02024       // Avoid image buttons (which come with garbage value), so we only get value
02025       // for the button actually clicked.
02026       if (!isset($element['#value']) && empty($element['#has_garbage_value'])) {
02027         $element['#value'] = isset($element['#default_value']) ? $element['#default_value'] : '';
02028       }
02029     }
02030   }
02031 
02032   // Determine which element (if any) triggered the submission of the form and
02033   // keep track of all the clickable buttons in the form for
02034   // form_state_values_clean(). Enforce the same input processing restrictions
02035   // as above.
02036   if ($process_input) {
02037     // Detect if the element triggered the submission via Ajax.
02038     if (_form_element_triggered_scripted_submission($element, $form_state)) {
02039       $form_state['triggering_element'] = $element;
02040     }
02041 
02042     // If the form was submitted by the browser rather than via Ajax, then it
02043     // can only have been triggered by a button, and we need to determine which
02044     // button within the constraints of how browsers provide this information.
02045     if (isset($element['#button_type'])) {
02046       // All buttons in the form need to be tracked for
02047       // form_state_values_clean() and for the form_builder() code that handles
02048       // a form submission containing no button information in $_POST.
02049       $form_state['buttons'][] = $element;
02050       if (_form_button_was_clicked($element, $form_state)) {
02051         $form_state['triggering_element'] = $element;
02052       }
02053     }
02054   }
02055 
02056   // Set the element's value in $form_state['values'], but only, if its key
02057   // does not exist yet (a #value_callback may have already populated it).
02058   if (!drupal_array_nested_key_exists($form_state['values'], $element['#parents'])) {
02059     form_set_value($element, $element['#value'], $form_state);
02060   }
02061 }

02103 function _form_button_was_clicked($element, &$form_state) {
02104   // First detect normal 'vanilla' button clicks. Traditionally, all
02105   // standard buttons on a form share the same name (usually 'op'),
02106   // and the specific return value is used to determine which was
02107   // clicked. This ONLY works as long as $form['#name'] puts the
02108   // value at the top level of the tree of $_POST data.
02109   if (isset($form_state['input'][$element['#name']]) && $form_state['input'][$element['#name']] == $element['#value']) {
02110     return TRUE;
02111   }
02112   // When image buttons are clicked, browsers do NOT pass the form element
02113   // value in $_POST. Instead they pass an integer representing the
02114   // coordinates of the click on the button image. This means that image
02115   // buttons MUST have unique $form['#name'] values, but the details of
02116   // their $_POST data should be ignored.
02117   elseif (!empty($element['#has_garbage_value']) && isset($element['#value']) && $element['#value'] !== '') {
02118     return TRUE;
02119   }
02120   return FALSE;

02141 function form_state_values_clean(&$form_state) {
02142   // Remove internal Form API values.
02143   unset($form_state['values']['form_id'], $form_state['values']['form_token'], $form_state['values']['form_build_id'], $form_state['values']['op']);
02144 
02145   // Remove button values.
02146   // form_builder() collects all button elements in a form. We remove the button
02147   // value separately for each button element.
02148   foreach ($form_state['buttons'] as $button) {
02149     // Remove this button's value from the submitted form values by finding
02150     // the value corresponding to this button.
02151     // We iterate over the #parents of this button and move a reference to
02152     // each parent in $form_state['values']. For example, if #parents is:
02153     //   array('foo', 'bar', 'baz')
02154     // then the corresponding $form_state['values'] part will look like this:
02155     // array(
02156     //   'foo' => array(
02157     //     'bar' => array(
02158     //       'baz' => 'button_value',
02159     //     ),
02160     //   ),
02161     // )
02162     // We start by (re)moving 'baz' to $last_parent, so we are able unset it
02163     // at the end of the iteration. Initially, $values will contain a
02164     // reference to $form_state['values'], but in the iteration we move the
02165     // reference to $form_state['values']['foo'], and finally to
02166     // $form_state['values']['foo']['bar'], which is the level where we can
02167     // unset 'baz' (that is stored in $last_parent).
02168     $parents = $button['#parents'];
02169     $last_parent = array_pop($parents);
02170     $key_exists = NULL;
02171     $values = &drupal_array_get_nested_value($form_state['values'], $parents, $key_exists);
02172     if ($key_exists && is_array($values)) {
02173       unset($values[$last_parent]);
02174     }
02175   }
02176 }

02239 function form_type_checkbox_value($element, $input = FALSE) {
02240   if ($input === FALSE) {
02241     // Use #default_value as the default value of a checkbox, except change
02242     // NULL to 0, because _form_builder_handle_input_element() would otherwise
02243     // replace NULL with empty string, but an empty string is a potentially
02244     // valid value for a checked checkbox.
02245     return isset($element['#default_value']) ? $element['#default_value'] : 0;
02246   }
02247   else {
02248     // Checked checkboxes are submitted with a value (possibly '0' or ''):
02249     // http://www.w3.org/TR/html401/interact/forms.html#successful-controls.
02250     // For checked checkboxes, browsers submit the string version of
02251     // #return_value, but we return the original #return_value. For unchecked
02252     // checkboxes, browsers submit nothing at all, but
02253     // _form_builder_handle_input_element() detects this, and calls this
02254     // function with $input=NULL. Returning NULL from a value callback means to
02255     // use the default value, which is not what is wanted when an unchecked
02256     // checkbox is submitted, so we use integer 0 as the value indicating an
02257     // unchecked checkbox. Therefore, modules must not use integer 0 as a
02258     // #return_value, as doing so results in the checkbox always being treated
02259     // as unchecked. The string '0' is allowed for #return_value. The most
02260     // common use-case for setting #return_value to either 0 or '0' is for the
02261     // first option within a 0-indexed array of checkboxes, and for this,
02262     // form_process_checkboxes() uses the string rather than the integer.
02263     return isset($input) ? $element['#return_value'] : 0;
02264   }
02265 }

04431 function batch_process($redirect = NULL, $url = 'batch', $redirect_callback = 'drupal_goto') {
04432   $batch =& batch_get();
04433 
04434   drupal_theme_initialize();
04435 
04436   if (isset($batch)) {
04437     // Add process information
04438     $process_info = array(
04439       'current_set' => 0,
04440       'progressive' => TRUE,
04441       'url' => $url,
04442       'url_options' => array(),
04443       'source_url' => $_GET['q'],
04444       'redirect' => $redirect,
04445       'theme' => $GLOBALS['theme_key'],
04446       'redirect_callback' => $redirect_callback,
04447     );
04448     $batch += $process_info;
04449 
04450     // The batch is now completely built. Allow other modules to make changes
04451     // to the batch so that it is easier to reuse batch processes in other
04452     // environments.
04453     drupal_alter('batch', $batch);
04454 
04455     // Assign an arbitrary id: don't rely on a serial column in the 'batch'
04456     // table, since non-progressive batches skip database storage completely.
04457     $batch['id'] = db_next_id();
04458 
04459     // Move operations to a job queue. Non-progressive batches will use a
04460     // memory-based queue.
04461     foreach ($batch['sets'] as $key => $batch_set) {
04462       _batch_populate_queue($batch, $key);
04463     }
04464
